async function generateKeywords() {
  const niche = nicheEl.value.trim();

  if (!niche) { showToast('⚠ Please describe your business or niche'); nicheEl.focus(); return; }

  document.getElementById('resultsWrap').style.display = 'none';

  startProgress();

  const systemPrompt = `You are an expert SEO strategist specialising in keyword research for new websites with low domain authority. Your job is to find keywords that are genuinely winnable for brand-new sites.

Return ONLY a valid JSON object. No markdown, no explanation, no backticks. The JSON must have this exact structure:
{
  "keywords": [
    {
      "keyword": "exact search phrase",
      "intent": "Informational" | "Commercial" | "Transactional",
      "difficulty": "Low" | "Medium" | "High",
      "contentType": "How-to guide" | "Listicle" | "Product showcase" | "Case study" | "Tool comparison" | "FAQ" | "Opinion piece" | "Tutorial" | "Review" | "Framework post",
      "isQuickWin": true | false
    }
  ]
}

Rules:
- Return exactly 20 keywords
- Focus on long-tail, low-competition keywords (3–6 words ideally)
- Bias heavily toward Low difficulty — new sites cannot compete for high-difficulty terms
- Mark the 5 best opportunities as isQuickWin: true (these are the absolute lowest competition relative to the niche)
- Keywords must be specific to the niche provided — no generic results
- Mix of intents but lean Informational and Commercial Investigation
- Content types must match the intent and keyword naturally`;

  const userPrompt = `Business/niche: ${niche}
Target audience: ${audience || 'general audience'}
Content goal: ${selectedGoal || 'drive traffic and grow the site'}

Generate 20 low-competition keywords this site can realistically rank for as a brand new domain. Focus on specificity to this exact niche.`;

  try {
    const response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': ANTHROPIC_API_KEY,
        'anthropic-version': '2023-06-01',
        'anthropic-dangerous-direct-browser-access': 'true'
      },
      body: JSON.stringify({
        model: 'claude-sonnet-4-6',
        max_tokens: 2000,
        system: systemPrompt,
        messages: [{ role: 'user', content: userPrompt }]
      })
    });

    if (!response.ok) {
      const err = await response.json().catch(() => ({}));
      throw new Error(err.error?.message || `API error ${response.status}`);
    }

    const data = await response.json();
    const raw = data.content.map(c => c.text || '').join('');
    const clean = raw.replace(/```json|```/g, '').trim();
    const parsed = JSON.parse(clean);

    generatedKeywords = parsed.keywords || [];
    stopProgress(true);
    renderResults(generatedKeywords);
    bumpCounter();

    // Scroll to results
    setTimeout(() => {
      document.getElementById('resultsWrap').scrollIntoView({ behavior: 'smooth', block: 'start' });
    }, 400);

  } catch (err) {
    console.error('Generation error:', err);
    stopProgress(false);
    showError();
  } finally {
    btn.disabled = false;
    btn.textContent = 'Generate My 20 Keywords →';
  }
}

/* ── COPY HELPERS ── */
function copyKw(text) {
  navigator.clipboard.writeText(text).then(() => showToast('Copied: ' + text)).catch(() => showToast('Keyword: ' + text));
}

function copyAll() {
  if (!generatedKeywords.length) return;
  const text = generatedKeywords.map(k => k.keyword).join('\n');
  navigator.clipboard.writeText(text).then(() => showToast('✓ All ' + generatedKeywords.length + ' keywords copied!')).catch(() => {});
}

function exportCSV() {
  if (!generatedKeywords.length) return;
  const header = 'Keyword,Intent,Difficulty,Content Type,Quick Win';
  const rows = generatedKeywords.map(k => `"${k.keyword}","${k.intent}","${k.difficulty}","${k.contentType}","${k.isQuickWin ? 'Yes' : 'No'}"`);
  const csv = [header, ...rows].join('\n');
  const blob = new Blob([csv], { type: 'text/csv' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'seo-keywords.csv';
  a.click();
  URL.revokeObjectURL(url);
  showToast('✓ CSV downloaded');
}
